[Coding Practice] 햄버거 만들기

햄버거 만들기 문제에 대한 설명은 프로그래머스 사이트를 참조바랍니다.

이 문제는 특정 패턴이 존재할 때마다 answer에 1씩 더해주면 되는 것이기 때문에 처음에는 아래 코드와 같이 문자열로 바꿔서 해당 패턴이 있을 때마다 찾아서 제거하는 방식으로 진행하였습니다. 그러나 아무리 효율적으로 하려고 해도 테스트 케이스 1개가 마지막까지 통과되지 못해서 할 수 없이 다른 방법을 생각해보기로 하였습니다.

import java.util.*;

class Solution {
    public int solution(int[] ingredient) {
        int answer = 0;
        // ingredient를 String으로 만든 후 "1231" 패턴이 나올 때마다 substring으로 제거.
        StringBuilder ingStr = new StringBuilder("");
        for (int i = 0; i < ingredient.length; i++) {
            ingStr.append(String.valueOf(ingredient[i]));
        }
        
        // 1000000 이므로 이미 한 번 확인한 index 중 앞의 몇개 이외에는 
        // 다시 확인하지 않도록 작성한다.
        int idx = 0;
        outerloop : while (true) {
            if (ingStr.length() < 4) {
                break;
            }

            for (int j = idx; j < ingStr.length() - 3; j++) {
                if (ingStr.substring(j, j + 4).equals("1231")) {
                    answer++;
                    int tmpIdx = ingStr.indexOf("1231");
                    ingStr.delete(tmpIdx, tmpIdx + 4);
                    if (j - 3 >= 0) {
                        idx = j - 3;
                    }
                    else {
                        idx = 0;
                    }
                    break;
                }
                if (!ingStr.substring(j, j + 4).equals("1231")
                        && j == ingStr.length() - 4) {
                    break outerloop;
                }
            }
        }

        return answer;
    }
}

위의 코드는 아래와 같이 아예 새로 변경되었습니다. 반복문을 여러 번 사용하지 않고 한 번에 처리하려면 스택 같은 구조가 필요했는데, 이번 문제는 한 개만 peek() 하는 것이 아닌 4개를 한꺼번에 점검해야 하기 때문에 ArrayList를 스택과 비슷한 방식으로 사용하였습니다. 반복문을 한 번 돌면서 그때그때 값을 리스트에 저장하다가 리스트 요소의 개수가 4개 이상일 때부터 특정 패턴이 나오는지 계속 확인을 하고 패턴이 존재하면 바로 요소 4개를 제거하면 되겠습니다.

import java.util.*;

class Solution {
    public int solution(int[] ingredient) {
        int answer = 0;
        // 시간 초과로 인해 스택을 사용하여 해결하는 방법으로 변경.
        // 한 번에 4개의 요소를 점검해야 하므로 Stack의 peek()이 아닌 
        // ArrayList를 Stack과 유사하게 사용.
        ArrayList<Integer> ingList = new ArrayList<>();
        for (int i = 0; i < ingredient.length; i++) {
            ingList.add(ingredient[i]);
            if (ingList.size() >= 4) {
                if (ingList.get(ingList.size() - 4) == 1 
                    && ingList.get(ingList.size() - 3) == 2
                    && ingList.get(ingList.size() - 2) == 3 
                    && ingList.get(ingList.size() - 1) == 1) {
                    answer++;
                    ingList.remove(ingList.size() - 1);
                    ingList.remove(ingList.size() - 1);
                    ingList.remove(ingList.size() - 1);
                    ingList.remove(ingList.size() - 1);
                }
            }
        }
        
        return answer;
    }
}
